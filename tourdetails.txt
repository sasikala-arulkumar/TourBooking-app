import React, { useEffect, useState } from 'react';
import { useParams, useNavigate } from 'react-router-dom';
import axios from 'axios';
import './TourDetails.css';

const TourDetails = () => {
  const { id } = useParams();
  const [tour, setTour] = useState(null);
  const [loading, setLoading] = useState(true);
   const navigate = useNavigate();

  useEffect(() => {
    axios.get(`http://localhost:3500/tours/${id}`)
      .then(res => {
        setTour(res.data);
        setLoading(false);
      })
      .catch(err => {
        console.error("Failed to fetch tour:", err);
        setLoading(false);
      });
  }, [id]);
  const handleBookNow = () => {
    navigate(`/book/${id}`); // ‚¨ÖÔ∏è navigate to booking page with tour id
  };

  if (loading) return <p>Loading tour details...</p>;
  if (!tour) return <p>No tour found for ID {id}</p>;

  return (
    <div className='head'>
    <div className="tour-detail-container">
      <img src={tour.image} alt={tour.title} className="detail-image" />
      <div className="tour-content">
        <h2>{tour.name}</h2>
        <p>üìç <strong>{tour.location}</strong></p>
        {/* <h3>Places include:</h3>
        <ul>
          <li>{tour.place1}</li>
        </ul> */}
        {tour.places && tour.places.length > 0 && (
  <div className="places-list">
    <h4>Places Covered:</h4>
    <ul>
      {tour.places.map((place, index) => (
        <li key={index}>{place}</li>
      ))}
    </ul>
  </div>
)}


        <p>{tour.description}</p>
        <h3 style={{ color: 'orangered' }}>Price: ‚Çπ{tour.price}</h3>
        <button className="book-now-button" onClick={handleBookNow}>
          Book This Tour
        </button>
      </div>
    </div>
    </div>
  );
};

export default TourDetails;
